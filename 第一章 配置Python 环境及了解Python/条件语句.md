# if 语句
	if 条件表达式:
		.....
	eg:
	isSuccess=True
	if isSuccess:
		print ('is true')

# if else 语句
	if 条件表达式:
		.....
	else:
		....
	eg:
	isSuccess:True
	if isSuccess:
		print ('is true')
	else:
		print ('is false')
# for 语句
	使用for ..... in ..结构 进行遍历元素
	eg:
	for i in [1,2,3,4,5]:
		print (i)
# while 循环
   eg:
   i=10
   while i>0:
        print (i)
        i=i-1
# 函数
	使用关键字 def 进行定义函数
	eg：
	def fun():
		print ('hello world')
